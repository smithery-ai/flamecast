Flamecast ships with a reference React UI, but you can build your own using the same hooks. The `useFlamecastSession` hook manages WebSocket connections, event buffering, and state synchronization so you can focus on rendering.

## The useFlamecastSession hook

The hook takes a session ID, fetches the session metadata to get the WebSocket URL, connects automatically, and returns everything you need to build an interactive UI.

```typescript
import { useFlamecastSession } from "@flamecast/sdk/client/hooks/use-flamecast-session";

function SessionView({ sessionId }: { sessionId: string }) {
  const {
    events,            // All session events (readonly SessionLog[])
    connectionState,   // "disconnected" | "connecting" | "connected" | "reconnecting"
    isConnected,       // Shorthand for connectionState === "connected"
    prompt,            // (text: string) => void
    respondToPermission, // (requestId: string, body) => void
    cancel,            // (queueId?: string) => void
    terminate,         // () => void
    requestFilePreview, // (path: string) => Promise<{content, truncated, maxChars}>
  } = useFlamecastSession(sessionId);
}
```

<Note>
  The hook requires `@tanstack/react-query` to be set up with a `QueryClientProvider` in your app.
</Note>

## Render a conversation

Session events are raw log entries. Use `sessionLogsToSegments` to convert them into user messages, assistant messages, and tool calls:

```typescript
import { useMemo } from "react";
import { sessionLogsToSegments } from "@flamecast/sdk/client/lib/logs-markdown";
import { useFlamecastSession } from "@flamecast/sdk/client/hooks/use-flamecast-session";

function Conversation({ sessionId }: { sessionId: string }) {
  const { events, isConnected, prompt } = useFlamecastSession(sessionId);

  const segments = useMemo(() => sessionLogsToSegments(events), [events]);

  return (
    <div>
      {segments.map((seg, i) => {
        if (seg.kind === "user") {
          return <div key={i} className="user-message">{seg.text}</div>;
        }
        if (seg.kind === "assistant") {
          return <div key={i} className="assistant-message">{seg.text}</div>;
        }
        if (seg.kind === "tool") {
          return (
            <div key={i} className="tool-call">
              {seg.title} — {seg.status}
            </div>
          );
        }
        return null;
      })}
    </div>
  );
}
```

The segments returned are:

| Kind | Fields | Description |
|---|---|---|
| `user` | `text` | Concatenated user message text |
| `assistant` | `text` | Concatenated agent message chunks |
| `tool` | `toolCallId`, `title`, `status` | Tool call with live status updates |

## Send prompts

The `prompt` function sends a message to the agent over WebSocket:

```typescript
function PromptInput({ sessionId }: { sessionId: string }) {
  const [text, setText] = useState("");
  const { prompt, isConnected } = useFlamecastSession(sessionId);

  const handleSubmit = () => {
    if (!text.trim()) return;
    prompt(text);
    setText("");
  };

  return (
    <div>
      <input
        value={text}
        onChange={(e) => setText(e.target.value)}
        onKeyDown={(e) => e.key === "Enter" && handleSubmit()}
        disabled={!isConnected}
        placeholder="Send a prompt..."
      />
      <button onClick={handleSubmit} disabled={!isConnected}>
        Send
      </button>
    </div>
  );
}
```

## Handle permission requests

When an agent requests permission, it shows up in the event stream. Derive the pending permission from events and render approval buttons:

```typescript
function PermissionCard({ sessionId }: { sessionId: string }) {
  const { events, respondToPermission } = useFlamecastSession(sessionId);

  // Find the latest unresolved permission request
  const pendingPermission = useMemo(() => {
    for (let i = events.length - 1; i >= 0; i--) {
      const event = events[i];
      if (
        event.type === "permission_approved" ||
        event.type === "permission_rejected" ||
        event.type === "permission_cancelled"
      ) {
        return null;
      }
      if (event.type === "permission_request" && event.data.pendingPermission) {
        return event.data.pendingPermission;
      }
    }
    return null;
  }, [events]);

  if (!pendingPermission) return null;

  return (
    <div className="permission-card">
      <p>{pendingPermission.title}</p>
      {pendingPermission.options.map((opt) => (
        <button
          key={opt.optionId}
          onClick={() =>
            respondToPermission(pendingPermission.requestId, {
              optionId: opt.optionId,
            })
          }
        >
          {opt.name}
        </button>
      ))}
      <button
        onClick={() =>
          respondToPermission(pendingPermission.requestId, {
            outcome: "cancelled",
          })
        }
      >
        Cancel
      </button>
    </div>
  );
}
```

## Create sessions from the UI

Use the REST client to list templates and create sessions:

```typescript
import { createFlamecastClient } from "@flamecast/sdk/client";
import { useQuery, useMutation } from "@tanstack/react-query";

const client = createFlamecastClient({ baseUrl: "http://localhost:3001" });

function AgentLauncher() {
  const { data: templates } = useQuery({
    queryKey: ["templates"],
    queryFn: () => client.fetchAgentTemplates(),
  });

  const createSession = useMutation({
    mutationFn: (templateId: string) =>
      client.createSession({ agentTemplateId: templateId }),
    onSuccess: (session) => {
      // Navigate to session view
      window.location.href = `/sessions/${session.id}`;
    },
  });

  return (
    <div>
      {templates?.map((t) => (
        <button key={t.id} onClick={() => createSession.mutate(t.id)}>
          {t.name}
        </button>
      ))}
    </div>
  );
}
```

## Browse agent files

Use `requestFilePreview` to read files from the agent's workspace:

```typescript
function FileViewer({ sessionId }: { sessionId: string }) {
  const { requestFilePreview, isConnected } = useFlamecastSession(sessionId);
  const [content, setContent] = useState<string | null>(null);

  const loadFile = async (path: string) => {
    const result = await requestFilePreview(path);
    setContent(result.content);
  };

  return (
    <div>
      <button onClick={() => loadFile("/src/index.ts")} disabled={!isConnected}>
        Load index.ts
      </button>
      {content && <pre>{content}</pre>}
    </div>
  );
}
```

## Full example

Putting it all together — a minimal but complete agent UI:

```typescript
import { useState, useMemo } from "react";
import { QueryClient, QueryClientProvider } from "@tanstack/react-query";
import { useFlamecastSession } from "@flamecast/sdk/client/hooks/use-flamecast-session";
import { sessionLogsToSegments } from "@flamecast/sdk/client/lib/logs-markdown";

const queryClient = new QueryClient();

function App({ sessionId }: { sessionId: string }) {
  return (
    <QueryClientProvider client={queryClient}>
      <AgentChat sessionId={sessionId} />
    </QueryClientProvider>
  );
}

function AgentChat({ sessionId }: { sessionId: string }) {
  const [input, setInput] = useState("");
  const {
    events,
    isConnected,
    prompt,
    respondToPermission,
  } = useFlamecastSession(sessionId);

  const segments = useMemo(() => sessionLogsToSegments(events), [events]);

  const pendingPermission = useMemo(() => {
    for (let i = events.length - 1; i >= 0; i--) {
      const event = events[i];
      if (
        event.type === "permission_approved" ||
        event.type === "permission_cancelled"
      ) return null;
      if (event.type === "permission_request") {
        return event.data.pendingPermission;
      }
    }
    return null;
  }, [events]);

  return (
    <div>
      <div className="status">
        {isConnected ? "Connected" : "Connecting..."}
      </div>

      <div className="messages">
        {segments.map((seg, i) => (
          <div key={i} className={seg.kind}>
            {seg.kind === "tool"
              ? `${seg.title} — ${seg.status}`
              : seg.text}
          </div>
        ))}
      </div>

      {pendingPermission && (
        <div className="permission">
          <p>{pendingPermission.title}</p>
          {pendingPermission.options.map((opt) => (
            <button
              key={opt.optionId}
              onClick={() =>
                respondToPermission(pendingPermission.requestId, {
                  optionId: opt.optionId,
                })
              }
            >
              {opt.name}
            </button>
          ))}
        </div>
      )}

      <div className="input">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => {
            if (e.key === "Enter" && input.trim()) {
              prompt(input);
              setInput("");
            }
          }}
          placeholder="Send a prompt..."
        />
      </div>
    </div>
  );
}
```
